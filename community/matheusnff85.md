class Developer:
    def __init__(self, name, city, age, hobbies):
        self.name = name
        self.city = city
        self.age = age
        self.hobbies = hobbies

    def __str__(self):
        return (
          f"My Name is: {self.name}\n"
          f"I live in {self.city}\n"
          f"I am {self.age} years old\n"
          f"Hobbies: {', '.join(self.hobbies)}.\n"
        )


me = Developer(
    "Matheus",
    "Nova Friburgo - RJ",

    ["Games", "Music", "Series", "Soccer"]
)
